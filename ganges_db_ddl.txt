CREATE TABLE RETAILER_BILL_HEADER ( BILL_HEADER_UUID VARCHAR(72) NOT NULL , STORE_UUID VARCHAR(72) NOT NULL , BILL_DATE TIMESTAMP NOT NULL , BILL_DISCOUNT DOUBLE NOT NULL , BILL_NUM VARCHAR(72) NOT NULL , BILL_TOTAL DOUBLE NOT NULL , CREATED_BY VARCHAR(72), CREATED_TIME TIMESTAMP NOT NULL , REF_COLUMN_1 VARCHAR(30), REF_COLUMN_2 VARCHAR(30), UPDATE_BY VARCHAR(72), UPDATED_TIME TIMESTAMP NOT NULL , BILL_STATUS INTEGER, PRIMARY KEY (BILL_HEADER_UUID,STORE_UUID));
